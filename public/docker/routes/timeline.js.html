exports.show = function (req, res) {
  PostSchema.findAll({
    where: ["username = ?", req.session.username]}
       ).on('success', function(newPost){
    res.render('timeline', {  title : 'Timeline',
                  newPost: newPost
              });
  })
};

exports.showDiscover = function (req, res) {
  PostSchema.findAll(
  {
         where: ["id = ?", 1]

              });

};

exports.followUser = function(req, res){
    var username = req.session.username,
        following = req.param('following');
        if(username == following){
          req.flash('followingError', 'You can\'t\ follow yourself');
        }
        else if(!following){
          req.flash('followingError', 'For the safety purpose, please re-type username correctly');          
        }
        followUser.find({
          where: {
            following : following
          }
        })
        .success(function(user){
          if(user){
            req.flash('followingError', 'You are already following this user');
          }
          else{
            followUser.build({
            username: req.session.username,
            following: following,
            id: true
            })
            .save();
          }
        })
    res.redirect('/following');
};

exports.unfollow = function(req, res){
    var username = req.session.username,
        following = req.param('following');
        if(!following){
          req.flash('followingError', 'For the safety purpose, please re-type username correctly');
        }
        else{
        followUser.find({
          where: {
            following : following
          }
        })
        .success(function(user){
          if(user){
            var task = followUser.build({
            username: req.session.username,
            following: following,
            id: true
            });
            task.id = false;
            task.save(['id']);
          }
          else{
            req.flash('followingError', 'For the safety purpose, please re-type username correctly');
          }
          res.redirect('/following');
       });
    }
};

exports.showFollowers = function(req,res){
    followUser.findAll({
        where: ["username = ? AND id=?", req.session.username, 1]}
           ).on('success', function(following){
        res.render('following', {  title : 'Whom you are following',
                      following : following
                  });
  })
}
